-- County Financial Database Schema
CREATE TABLE projects (
  id INT PRIMARY KEY AUTO_INCREMENT,
  project_name VARCHAR(255) NOT NULL,
  budget DECIMAL(15,2),
  expenditure DECIMAL(15,2) DEFAULT 0,
  department VARCHAR(100),
  status ENUM('active','completed','pending'),
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Budget vs Expenditure Report
SELECT p.project_name, p.budget,
       SUM(e.amount) AS total_spent,
       (p.budget - SUM(e.amount)) AS balance
FROM projects p
LEFT JOIN expenditures e ON p.id = e.project_id
GROUP BY p.id;
